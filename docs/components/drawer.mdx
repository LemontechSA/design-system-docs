## Import

```js
import { Drawer } from "@suit-ui/react";
```

## Usage

import { useState } from "react";
import { Box, Button, Drawer } from "@suit-ui/react";

export const DrawerExample = () => {
  const [isOpen, setIsOpen] = useState(false);
  return (
    <Box>
      <Button onClick={() => setIsOpen(true)}>Open drawer</Button>
      <Drawer
        className="z-[99999]"
        drawerClassName="z-[99999]"
        isOpen={isOpen}
        onClose={() => setIsOpen(false)}
      >
        <Drawer.CloseButton />
        <Drawer.Brand>Suit UI</Drawer.Brand>
        <Drawer.Link isExternal href="https://www.google.com">
          Go to google
        </Drawer.Link>
        <Drawer.Link>Home</Drawer.Link>
        <Drawer.Link>About us</Drawer.Link>
        <Drawer.Link>Contact</Drawer.Link>
      </Drawer>
    </Box>
  );
};
